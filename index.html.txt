<script>
function showDay(day){
    const workouts = {
        Martedì: ["Panca 5x5", "Military 4x6", "Push Press 4x5", "Push-up esplosivi"],
        Giovedì: ["Stacco 5x5", "Rematore 4x8", "Farmer walk"],
        Sabato: ["Push", "Legs", "Pull"]
    };
    document.getElementById("workout").innerHTML =
    "div class='card'><h2>" +workouts[day] + "</h2></div>";
}

let interval;
function startTimer(seconds){
    clearInterval(interval);
    let time = seconds;
    document.getElementById("timer").innerText = time;
    interval = setInterval(() => {
        time--;
        document.getElementById("timer").innerText = time;
        if(time <= 0){
            clearInterval(interval);
            alert("Round finito!");
        }
    }, 1000);
}

function savedata(){
    let ex= document.getElementById("exercise").value;
    let w = document.getElementById("weight").value;

    let data = JSON.parse(localStorage.getItem("log")) || [];
    data.push(ex + ": " + w + "kg");

    localStorage.set.Item("log", JSON.stringify(data));
    showHistory();
}

function showHistory(){
    let data = JSON.parse(localStorage.getItem("log")) || [];
    document.getElementById("history").innerHTML = data.join("<br>");
}

showHistory();
